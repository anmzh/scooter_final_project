SELECT c.login, COUNT(o.id) AS total_order FROM "Orders" AS o INNER JOIN "Couriers" AS c ON o."courierId" = c.id WHERE o."inDelivery" = true GROUP BY c.login;
SELECT track, CASE WHEN finished = true THEN '2' WHEN cancelled = true THEN '-1' WHEN "inDelivery" = true THEN '1' ELSE '0' END AS order_status FROM "Orders";
